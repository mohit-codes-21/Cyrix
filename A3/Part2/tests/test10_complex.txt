struct Point {
    int x;
    int y;
};

int factorial(int n) {
    if (n <= 1)
        return 1;
    else
        return n * factorial(n - 1);
}

int compute(int a, int b, int mode = 0) {
    switch (mode) {
        case 0: return a + b;
        case 1: return a - b;
        default: return 0;
    }
}

int main() {
    struct Point p;
    p.x = 10;
    p.y = 20;

    int result = factorial(5);

    for (int i = 0; i < 10; i++) {
        result = result + compute(p.x, p.y);
    }

    var z = result * 2;

    int sum = 0;
    while (sum < 100) {
        sum = sum + 1;
        if (sum > 50)
            break;
    }

    return result;
}
