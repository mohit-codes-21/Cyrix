// --- Unterminated string (no closing quote) ---
// The lexer should handle partial strings gracefully.
// Note: can't test unterminated string in same file without breaking
// subsequent tests, so we test strings at boundaries instead.

// --- Empty char constant '' (invalid) ---
// This should NOT match the char constant rule.
// '' should produce two UNKNOWN tokens (or two separate quotes)

// --- 0x with no hex digits (invalid hex) ---
// Should be: I_CONSTANT 0, then IDENTIFIER x
0x
0X

// --- 0b101 (binary literal - not supported) ---
// Should be: I_CONSTANT 0, IDENTIFIER b101
0b101

// --- Numbers immediately followed by identifiers ---
// 123abc should be: I_CONSTANT 123, IDENTIFIER abc
123abc
0xFFghi

// --- Deeply nested parens and brackets ---
(((((((((())))))))))
[[[[]]]]

// --- Multiple operators in sequence ---
+-*/
!~&|
<><=>=

// --- Equal signs chaining ---
a=b=c=d

// --- Very long chain of dots ---
a.b.c.d.e.f.g

// --- Mixed safe-dot and regular dot ---
a?.b.c?.d.e?.f

// --- Three-way comparison in expression context ---
(a <=> b) + (c <=> d)

// --- Power in complex expression ---
a**b**c

// --- Safe deref of function result ---
func()?.field

// --- Lambda in assignment ---
f = () => 0;
g = (x, y) => x + y;

// --- Foreach over range ---
foreach (i in range) {
    match(i) {
        case 0: break;
        default: continue;
    }
}
