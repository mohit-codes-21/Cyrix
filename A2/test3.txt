// ---------------------------------------------------------
// SECTION 8: OPERATOR AMBIGUITY / MAXIMAL MUNCH
// ---------------------------------------------------------

// <=> should be TH_OP, not <= then >
a<=>b

// ** should be POW_OP, not * then *
a**b

// => should be LAMBDA_ARROW, not = then >
a=>b

// ?. should be SAFE_DOT, not ? then .
a?.b

// *** should be POW_OP then *
a***b

// ++ should be INC_OP, not + then +
a++

// -- should be DEC_OP, not - then -
a--

// -> should be PTR_OP, not - then >
a->b

// && should be AND_OP, not & then &
a&&b

// || should be OR_OP, not | then |
a||b

// <= should be LE_OP, not < then =
a<=b

// >= should be GE_OP, not > then =
a>=b

// == should be EQ_OP, not = then =
a==b

// != should be NE_OP, not ! then =
a!=b

// = followed by > (with space) is NOT lambda arrow
a = > b

// ? followed by . (with space) should be ? then .
a ? . b

// Stacking multiple ambiguous operators
a<=>b**c=>d?.e

// <=>> should be <=> then >
a<=>>b

// **= is POW_OP then =
a**=b

// ?.. is SAFE_DOT then .
a?..b
